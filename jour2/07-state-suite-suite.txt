ne fonctionne pas de la même manière si on utilise dans variable complexe 


const [p1 , setP1] = useState(["Alain" , "Céline"]); // je lui met un tableau 

const [p2, setP2] = useState({ nom : "Benoit" , age : 20 }) // je lui met un objet
